% Discourage (but don't forbid) widows/orphans. =10000 is infinite and forces
% LaTeX to eject whole paragraphs to the next page, leaving large mid-document
% gaps; a high *finite* penalty keeps pages filled while still avoiding most
% stray lines. Heading->table reserves use \Needspace* (see keep-headings filter).
\widowpenalty=9999
\clubpenalty=9999
\displaywidowpenalty=9999
\brokenpenalty=9999

% leftover space collects at the bottom rather than as gaps between blocks.

% Section styling - larger, colored
\usepackage{titlesec}
% Never break a page right after a heading. \@secpenalty (-300) makes a break
% *before* a heading attractive, which strands a heading that's immediately
% followed by another heading (or a table/figure block that won't fit). Prepending
% \nobreak to \@afterheading forbids the break, so the heading travels with its
% following content (sub-heading, paragraph, table, or figure).
\makeatletter
\preto\@afterheading{\nobreak}
\makeatother
\titleformat{\chapter}[display]
  {\normalfont\headingfont\fontsize{22}{26}\selectfont\color{questdb-heading}}
  {}
  {0pt}
  {}
\titlespacing*{\chapter}{0pt}{-30pt}{20pt}

\usepackage{needspace}
\titleformat{\section}
  {\normalfont\headingfont\fontsize{18}{22}\selectfont\color{questdb-heading}}

% Keep a *section* heading from stranding in the last lines of a page: reserve
% room for the heading plus a couple of following lines, else start a new page.
% Only sections get this. Subsections deliberately do NOT, so they flow and fill
% the page; \nobreak (above) already keeps any heading with its first line, which
% is enough to stop sub-headings dangling without forcing large bottom gaps.
\preto\section{\Needspace*{4\baselineskip}}

% Keep a code block with the lead-in paragraph that introduces it (e.g.
% "Example from XRP Ledger:"). Forbid a page break between the paragraph and the
% following code block so a lead-in never strands at the bottom of a page; if
% they do not fit together, both move to the next page.
\BeforeBeginEnvironment{Shaded}{\par\nobreak}
